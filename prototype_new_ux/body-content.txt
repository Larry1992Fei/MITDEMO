        function renderProgress() {
            const steps = getStepsForMode(currentMode);
            const ph = document.getElementById('progress-header');
            ph.innerHTML = steps.map((s, i) => {
                const state = i < currentStep ? 'completed' : i === currentStep ? 'active' : '';

        // ─── Action Bar ───
        function renderActionBar() {
            const steps = getStepsForMode(currentMode);
            const hints = getHintsForMode(currentMode);
            const btnLabels = getBtnLabelsForMode(currentMode);
            const isLast = currentStep === steps.length - 1;

            document.getElementById('action-hint').textContent = hints[currentStep] || '';
            document.getElementById('btn-back').style.display = currentStep > 0 ? 'inline-flex' : 'none';

            const btnNext = document.getElementById('btn-next');
            if (isLast) {
                btnNext.style.display = 'none';
            } else {
                btnNext.style.display = 'inline-flex';
                btnNext.textContent = btnLabels[currentStep] || '下一步 →';
                btnNext.className = 'btn ' + (currentStep >= 1 ? 'btn-success' : 'btn-primary');
            }
        }

        function updateDynamic() {

            const periodUnit = document.getElementById('p-periodUnit')?.value || 'M';

            const payment = getCurrentPayment(); // 获取当前支付方式

            // 商户自管模式参数（从 m-1 面板读取）

            const mAmount = mAmountInput ? mAmountInput.value : '9.99';
            const mCurrency = document.getElementById('m-currency')?.value || 'USD';
            const mSubject = document.getElementById('m-subject')?.value || '代扣标题';
            const mUserId = document.getElementById('m-userId')?.value || 'test1111111';
            
            // 调试：输出 mAmount 的值
            console.log('updateDynamic - mAmount:', mAmount);

            // Show/hide conditional rows
            document.getElementById('trialDaysRow').style.display = subtype === 'trial' ? 'flex' : 'none';
            document.getElementById('discountParamsRow').style.display = subtype === 'discount' ? 'block' : 'none';
            document.getElementById('trialDiscountParamsRow').style.display = subtype === 'trial-discount' ? 'block' : 'none';

            const setTxt = (id, val) => { const el = document.getElementById(id); if (el) el.textContent = val; };

            // PayerMax step 2 — subscriptionCreate
            setTxt('c-totalPeriods', totalPeriods);
            setTxt('c-periodUnit', periodUnit);
            setTxt('c-periodCount', periodCount);
            setTxt('c-amount', amount);
            setTxt('c-currency', currency);
            setTxt('c-startDate', startDate ? new Date(startDate).toISOString().replace('Z', '+08:00') : '—');

            // PayerMax step 3 — orderAndPay activate
            // 根据订阅类型计算激活金额
            let activateAmount = amount;
            if (subtype === 'trial') {
                // N天试用：激活金额为0（免费试用）
                activateAmount = '0';
            } else if (subtype === 'trial-discount') {
                // 试用+优惠：激活金额为试用期金额
                const trialAmountCombo = document.getElementById('p-trialAmount-combo')?.value || '0.99';
                activateAmount = trialAmountCombo;
            }

            setTxt('a-currency', currency);
            
            // 根据支付方式更新 paymentMethodType
            const paymentMethodTypeMap = {
                'card': 'CARD',
                'applepay': 'APPLEPAY',
                'googlepay': 'GOOGLEPAY',
                'apm': 'ONE_TOUCH' // APM 指定支付方式时使用 ONE_TOUCH，全量收银台时可不传
            };
            setTxt('a-paymentMethodType', paymentMethodTypeMap[payment] || 'CARD');

            // PayerMax step 4 — Webhook
            setTxt('w-amount', amount); setTxt('w-currency', currency);

            // PMX completion summary
            const unitMap = { D: '日', W: '周', M: '月', Y: '年' };
            setTxt('f-cycle', `每 ${periodCount} ${unitMap[periodUnit] || periodUnit} · 共 ${totalPeriods} 期`);
            setTxt('f-amount', `${currency} ${amount}`);
            setTxt('f-startDate', startDate ? startDate.replace('T', ' ') : '—');

            setTxt('m-bind-amount', mAmount);
            setTxt('m-bind-currency', mCurrency);
            setTxt('m-bind-subject', mSubject);
            setTxt('m-bind-userId', mUserId);
            setTxt('m-tradeNo', Date.now());

            setTxt('m-bind-amount-api', mAmount);
            setTxt('m-bind-currency-api', mCurrency);
            setTxt('m-bind-subject-api', mSubject);
            setTxt('m-bind-userId-api', mUserId);
            setTxt('m-tradeNo-api', Date.now());
            // paymentMethodType 动态更新
            setTxt('m-bind-paymentMethodType', merchantPaymentMethodType);
            setTxt('m-bind-paymentMethodType-api', merchantPaymentMethodType);

            // 根据集成方式切换商户自管模式第二步的代码块显示
            const integration = getCurrentIntegration();
            if (currentMode === 'merchant' && steps[currentStep]?.id === 'm-2') {
                const cashierBlock = document.querySelector('.m-bind-code-cashier');
                const apiBlock = document.querySelector('.m-bind-code-api');
                if (cashierBlock && apiBlock) {
                    if (integration === 'api') {
                        cashierBlock.style.display = 'none';
                        apiBlock.style.display = 'block';
                    } else {
                        cashierBlock.style.display = 'block';
                        apiBlock.style.display = 'none';
                    }
                }
            }

            // Merchant step 3 — deduct orderAndPay（使用 m-1 面板的参数）
            setTxt('m-dedAmount', mAmount);
            setTxt('m-dedCurrency', mCurrency);
            setTxt('m-dedTotalPeriods', totalPeriods);
            setTxt('m-dedTradeNo', Date.now());
            setTxt('m-dedSubNo', Date.now());
            setTxt('m-dedDeductNo', Date.now() + 2);
            setTxt('m-dedResAmt', mAmount);
            setTxt('m-dedResCur', mCurrency);
            // 添加 paymentMethodType 动态更新
            setTxt('m-deduct-paymentMethodType', merchantPaymentMethodType);

            // Component session request — update componentList based on payment method
            let componentListText = '';
            if (payment === 'card') {
                componentListText = '"CARD", "APPLEPAY", "GOOGLEPAY"';
            } else if (payment === 'applepay') {
                componentListText = '"APPLEPAY"';
            } else if (payment === 'googlepay') {
                componentListText = '"GOOGLEPAY"';
            }
            setTxt('s-componentList', componentListText);
            // Session 金额与激活金额保持一致
            setTxt('s-amount', activateAmount);
            setTxt('s-currency', currency);

            // Merchant mode session request — update componentList
            setTxt('sm-componentList', componentListText);
            // Merchant 模式的 session 金额使用 m-1 面板的参数
            setTxt('sm-amount', mAmount);
            setTxt('sm-currency', mCurrency);

            // Non-periodic mode — update parameters from np- fields
            const npAmount = document.getElementById('np-amount')?.value || '9.99';
            const npCurrency = document.getElementById('np-currency')?.value || 'USD';
            
            // Non-periodic mode session request — update componentList
            setTxt('snp-componentList', componentListText);
            setTxt('snp-amount', npAmount);
            setTxt('snp-currency', npCurrency);
            
            // 如果当前在 PayerMax 模式的第二步（创建订阅计划），自动切换代码块显示
            const steps = getStepsForMode(currentMode);
            if (currentMode === 'payermax' && steps[currentStep]?.id === 'pm-2') {
                switchCreateTab(subtype, true);
            }
            
            // 如果当前在商户自管模式的第二步（首次绑定支付方式），自动切换代码块显示
            if (currentMode === 'merchant' && steps[currentStep]?.id === 'm-2') {
                switchMerchantBindTab(subtype, true);
            }
        }

                        amount: trialAmount,
                        currency: currency // 使用主币种

        function copyCode(id) {
            const el = document.getElementById(id);
            if (!el) return;
            const raw = el.innerText || el.textContent;
            navigator.clipboard.writeText(raw).then(() => {
                const btn = el.closest('.code-block').querySelector('.copy-mini-btn');
                if (btn) { btn.textContent = '? 已复制'; setTimeout(() => btn.textContent = '复制', 2000); }
            });
